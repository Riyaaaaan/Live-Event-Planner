### Security Rules Example

```javascript
// Firestore Security Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Users collection
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create: if request.auth.uid == userId;
      allow update, delete: if request.auth.uid == userId
        || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Events collection
    match /events/{eventId} {
      allow read: if resource.data.visibility == 'public'
        || request.auth != null;
      allow create: if request.auth != null
        && request.auth.token.role in ['organizer', 'admin'];
      allow update, delete: if request.auth.uid == resource.data.organizerId
        || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';

      // Registrations subcollection
      match /registrations/{registrationId} {
        allow read: if request.auth.uid == resource.data.userId
          || request.auth.uid == get(/databases/$(database)/documents/events/$(eventId)).data.organizerId;
        allow create: if request.auth != null;
        allow update, delete: if request.auth.uid == resource.data.userId;
      }
    }
  }
}
```
